        function handleImageUpload(event, slotId) {
            const file = event.target.files[0];
            if (file) {
                const reader = new FileReader();
                reader.onload = function(e) {
                    const slotElement = document.getElementById(slotId).parentElement;
                    const uploadArea = slotElement.querySelector('.image-upload-area');
                    
                    uploadArea.innerHTML = `
                        <img src="${e.target.result}" alt="Uploaded food image" class="uploaded-image w-full h-full">
                    `;
                    uploadArea.classList.add('has-image');
                    uploadArea.onclick = () => triggerImageUpload(slotId);
                };
                reader.readAsDataURL(file);
            }
        }

        function saveAsPDF() {
            // Show loading message
            const originalText = event.target.textContent;
            event.target.textContent = 'กำลังสร้าง PDF...';
            event.target.disabled = true;
            
            // Hide all delete buttons before capturing
            const deleteButtons = document.querySelectorAll('.no-print');
            deleteButtons.forEach(btn => {
                btn.style.display = 'none';
            });
            
            // Get the document element
            const element = document.getElementById('documentContainer');
            
            // Use html2canvas to capture the element with better settings
            html2canvas(element, {
                scale: 3,
                useCORS: true,
                allowTaint: true,
                backgroundColor: '#ffffff',
                logging: false,
                width: 794, // A4 width in pixels at 96 DPI
                height: 1123, // A4 height in pixels at 96 DPI
                scrollX: 0,
                scrollY: 0,
                windowWidth: 794,
                windowHeight: 1123
            }).then(canvas => {
                // Show delete buttons again
                deleteButtons.forEach(btn => {
                    btn.style.display = '';
                });
                
                const imgData = canvas.toDataURL('image/jpeg', 0.95);
                
                // Create PDF using jsPDF with exact A4 dimensions
                const { jsPDF } = window.jspdf;
                const pdf = new jsPDF({
                    orientation: 'portrait',
                    unit: 'mm',
                    format: 'a4',
                    compress: true
                });
                
                // Add image to fit exactly on A4 page
                pdf.addImage(imgData, 'JPEG', 0, 0, 210, 297);
                
                // Generate filename with current date
                const today = new Date();
                const day = today.getDate().toString().padStart(2, '0');
                const month = (today.getMonth() + 1).toString().padStart(2, '0');
                const year = today.getFullYear() + 543;
                const filename = `รายงานอาหารกลางวัน_${day}-${month}-${year}.pdf`;
                
                // Save the PDF
                try {
                    pdf.save(filename);
                    alert('✅ บันทึกไฟล์ PDF เรียบร้อยแล้ว!\n\n📁 ไฟล์ถูกบันทึกในโฟลเดอร์ Downloads');
                } catch (saveError) {
                    console.error('Save error:', saveError);
                    // Try alternative download method
                    const blob = pdf.output('blob');
                    const url = URL.createObjectURL(blob);
                    const a = document.createElement('a');
                    a.href = url;
                    a.download = filename;
                    document.body.appendChild(a);
                    a.click();
                    document.body.removeChild(a);
                    URL.revokeObjectURL(url);
                    alert('✅ บันทึกไฟล์ PDF เรียบร้อยแล้ว!\n\n📁 ไฟล์ถูกบันทึกในโฟลเดอร์ Downloads');
                }
                
                // Reset button
                event.target.textContent = originalText;
                event.target.disabled = false;
                
            }).catch(error => {
                // Show delete buttons again in case of error
                deleteButtons.forEach(btn => {
                    btn.style.display = '';
                });
                
                console.error('Error generating PDF:', error);
                alert('❌ เกิดข้อผิดพลาดในการสร้าง PDF\n\nกรุณา:\n1. ตรวจสอบว่ามีรูปภาพในเอกสาร\n2. ลองรีเฟรชหน้าเว็บ\n3. ลองใหม่อีกครั้ง');
                
                // Reset button
                event.target.textContent = originalText;
                event.target.disabled = false;
            });
        }
